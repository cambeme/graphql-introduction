Type System

          
              type User {
                id: Int!
                name: String
                age: Int
              }
          
        
          
              type Query {
                getUser(id: Int!): User
                GetAllUsers: [User]
              }
          
        
          
              type Mutation {
                createUser(name: String!): User!
                updateUser(id: Int!, name: String!): User!
                deleteUser(id: Int!): Response!
              }
          
        
          
              type Subscription {
                userCreated: User
              }

Query 1/2

          
            query {
              GetAllUsers {
                id
                name
                age
              }
            }
          
        
          
            {
              "data": {
                "GetAllUsers": [
                {
                  "id": 1,
                  "name": "Stephen",

                },
                {
                  "id": 2,

                  "age": 28
                }
                ]
              }
            }

Query 2/2

          
            query ($id: Int!) {
              getUser(id: $id) {
                name
                age
              }
            }
          
        
          
            {
              "data": {
                "getUser": {
                  "name": "Stephen",
                  "age": 27
                }
              }
            }

Mutation


            mutation ($id: Int!, $name: String!) {
              updateUser(id: $id, name: $name) {
                id
                name
              }
            }
          

            {
              "data": {
                "updateUser": 
                  {
                   "id": 1,
                   "name": "Stephen new"
                  }
                ]
              }
            }
